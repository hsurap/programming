//Function to remove duplicates from sorted linked list.
Node *removeDuplicates(Node *head)
{
    if(head==NULL || head->next==NULL)
        return head;
    Node*st=head;
    Node*en=head->next;
    while(en!=NULL)
    {
        if(en->data==st->data)
        {
            Node*temp=en;
            en=en->next;
            delete temp;
        }
        else{
            st->next=en;
            st=en;
            en=en->next;
        }
    }
    st->next=NULL;
    return head;
}
